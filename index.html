function doPost(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sh=ss.getSheetByName("Sheet1")||ss.getActiveSheet(); var d=JSON.parse(e.postData.contents); if(d.action==="write"){ sh.appendRow([d.unit,d.week,d.group,d.amount,d.notes,d.timestamp]); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } if(d.action==="read"){ var rows=sh.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify({rows:rows})).setMimeType(ContentService.MimeType.JSON); } }
